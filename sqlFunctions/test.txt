create or replace function match_definitions (
  query_embedding vector(1536),
  match_threshold float,
  match_count int
)
returns table (
  id bigint,
  similarity float,
  code text,
  division text,
  title text,
  part text,
  chapter text,
  article text,
  section text,
  definitions text,
  similarity float
)
language sql stable
as $$
  select
    ca_code.id,
    1 - (ca_code.definition_embedding <=> query_embedding) as similarity,
    ca_code.code,
    ca_code.division,
    ca_code.title,
    ca_code.part,
    ca_code.chapter,
    ca_code.article,
    ca_code.section,
    ca_code.definitions,
    ca_code.definition_tokens
  from ca_code
  where ca_code.definition_embedding IS NOT NULL AND 1 - (ca_code.embedding <=> query_embedding) > match_threshold 
  order by similarity desc
  limit match_count;
$$;

create or replace function match_titles (
  query_embedding vector(1536),
  match_threshold float,
  match_count int
)
returns table (
  id bigint,
  definitions text,
  content text,
  similarity float
)
language sql stable
as $$
  select
    ca_code.id,
    ca_code.definitions,
    ca_code.content,
    1 - (ca_code.definition_embedding <=> query_embedding) as similarity
  from ca_code
  where 1 - (ca_code.definition_embedding <=> query_embedding) > match_threshold
  order by similarity desc
  limit match_count;
$$;
